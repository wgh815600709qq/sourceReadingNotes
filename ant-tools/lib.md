## Usage

```
require('colorful').colorful()
'hello'.to.red.color
'hello'.to.underline.bold.red.color

```

```
var Color = require('colorful').Color;
 
var s = new Color('colorful');
s.fgcolor = 13;
s.bgcolor = 61;
```
